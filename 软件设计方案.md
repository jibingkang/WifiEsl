## 一、项目概述

### 1.1 项目目标
开发一个Web应用软件，与WIFI标签系统API（Ver1.0.6）对接，实现：
- 设备信息状态实时查看
- 单个/批量设备控制
- 模板化屏幕内容更新（支持表单填写、导入导出）
- 实时状态推送和更新

### 2.2 技术选型

| 层级 | 技术 | 说明 |
|------|------|------|
| **前端框架** | Vue3 + TypeScript | 现代化、响应式、类型安全 |
| **UI组件库** | Element Plus | 企业级UI组件，表格/表单丰富 |
| **状态管理** | Pinia | 轻量级状态管理 |
| **实时通信** | MQTT.js + WebSocket | 前端MQTT连接 |
| **后端框架** | Python FastAPI | 高性能异步、自动文档 |
| **数据库** | SQLite + Redis(可选) | 轻量部署、缓存加速 |
| **任务队列** | Celery/BackgroundTasks | 异步批量操作 |
| **MQTT库** | paho-mqtt / asyncio-mqtt | Python MQTT客户端 |

#### 3.1.2 实时状态更新机制
```typescript
// composables/useMqtt.ts - MQTT实时通信组合函数
import { ref, onMounted, onUnmounted } from 'vue'
import { useMqttStore } from '@/stores/mqtt'
import type { MqttMessage } from '@/types/mqtt'

export function useMqtt() {
  const store = useMqttStore()
  const isConnected = ref(false)

  // 订阅的MQTT主题
  const topics = {
    // 设备状态
    online: '/client/${apiKey}/action/offline',
    offline: '/client/${apiKey}/action/offline',
    usbState: '/client/${apiKey}/action/usb_state',
    button: '/client/${apiKey}/action/button',
    
    // 操作反馈
    displayReply: '/client/${apiKey}/action/display_reply',
    batteryReply: '/client/${apiKey}/action/battery_reply',
    ledReply: '/client/${apiKey}/action/led_reply',
    rebootReply: '/client/${apiKey}/action/reboot_reply',
  }

  /**
   * 处理设备上线消息
   */
  const handleOnline = (message: MqttMessage) => {
    const { mac, msgId } = JSON.parse(message.payload)
    store.updateDeviceStatus(mac, 'online', msgId)
    
    // 触发UI更新事件
    window.dispatchEvent(new CustomEvent('device:status-change', {
      detail: { mac, status: 'online' }
    }))
  }

  /**
   * 处理设备下线消息
   */
  const handleOffline = (message: MqttMessage) => {
    const { mac, msgId } = JSON.parse(message.payload)
    store.updateDeviceStatus(mac, 'offline', msgId)
  }

  /**
   * 处理屏幕更新反馈
   */
  const handleDisplayReply = (message: MqttMessage) => {
    const { mac, msgId, result } = JSON.parse(message.payload)
    store.updateOperationResult('display', mac, result === 200)
    
    // 通知用户
    if (result === 200) {
      ElMessage.success(`设备 ${mac} 屏幕更新成功`)
    } else {
      ElMessage.error(`设备 ${mac} 屏幕更新失败`)
    }
  }

  /**
   * 处理电池电量反馈
   */
  const handleBatteryReply = (message: MqttMessage) => {
    const { mac, voltage } = JSON.parse(message.payload)
    store.updateDeviceField(mac, 'voltage', voltage)
  }

  onMounted(() => {
    // 连接MQTT并订阅主题
    store.connect()
    Object.values(topics).forEach(topic => {
      store.subscribe(topic)
    })
    
    // 注册消息处理器
    store.onMessage('online', handleOnline)
    store.onMessage('offline', handleOffline)
    store.onMessage('display_reply', handleDisplayReply)
    store.onMessage('battery_reply', handleBatteryReply)
  })

  onUnmounted(() => {
    // 断开连接
    store.disconnect()
  })

  return {
    isConnected,
    subscribe: store.subscribe,
    publish: store.publish,
  }
}
```

/**
 * 导出设备模板数据为Excel
 */
export function exportToExcel(
  devices: Array<{ mac: string; data: Record<string, any>; isCustom?: boolean }>,
  templateFields: TemplateFieldValue[],
  config: ExportConfig = { format: 'xlsx' }
): void {
  // 准备导出数据
  const headers = ['MAC地址', ...templateFields.map(f => f.label), '内容来源']
  
  const rows = devices.map(device => [
    device.mac,
    ...templateFields.map(f => device.data[f.key] ?? ''),
    device.isCustom ? '自定义' : '默认'
  ])

  const wsData = [headers, ...rows]
  const ws = XLSX.utils.aoa_to_sheet(wsData)

  // 设置列宽
  ws['!cols'] = [
    { wch: 22 },  // MAC地址
    ...templateFields.map(() => ({ wch: 20 })),
    { wch: 10 }   // 内容来源
  ]

  const wb = XLSX.utils.book_new()
  XLSX.utils.book_append_sheet(wb, ws, '设备更新数据')

  // 下载文件
  const filename = `设备模板数据_${formatDate(new Date())}.${config.format}`
  XLSX.writeFile(wb, filename)
}

/**
 * 从Excel导入设备数据
 */
export async function importFromExcel(
  file: File,
  templateFields: TemplateFieldValue[]
): Promise<{
  success: boolean
  data: ImportRow[]
  errors: Array<{ row: number; error: string }>
}> {
  return new Promise((resolve) => {
    const reader = new FileReader()
    const errors: Array<{ row: number; error: string }> = []

    reader.onload = (e) => {
      try {
        const data = new Uint8Array(e.target!.result as ArrayBuffer)
        const workbook = XLSX.read(data, { type: 'array' })
        const sheetName = workbook.SheetNames[0]
        const worksheet = workbook.Sheets[sheetName]
        const jsonData = XLSX.utils.sheet_to_json(worksheet) as Record<string, any>[]

        // 验证和处理每行数据
        const processedData: ImportRow[] = []

        jsonData.forEach((row, index) => {
          const rowNum = index + 2 // Excel行号从2开始（第1行是标题）

          // 验证必填字段MAC地址
          if (!row['MAC地址'] && !row['mac']) {
            errors.push({ row: rowNum, error: 'MAC地址不能为空' })
            return
          }

          // 构建标准化的数据对象
          const importRow: ImportRow = {
            mac: String(row['MAC地址'] || row['mac']).trim().toUpperCase()
          }

          // 映射模板字段
          templateFields.forEach(field => {
            // 尝试通过中文名或英文名匹配
            const value = row[field.label] || row[field.key]
            if (value !== undefined && value !== '') {
              importRow[field.key] = value
            }
          })

          // MAC地址格式验证
          if (!/^([0-9A-Fa-f]{2}:){5}[0-9A-Fa-f]{2}$/.test(importRow.mac)) {
            errors.push({ row: rowNum, error: `MAC地址格式错误: ${importRow.mac}` })
            return
          }

          processedData.push(importRow)
        })

        resolve({
          success: errors.length === 0,
          data: processedData,
          errors
        })
      } catch (error) {
        resolve({
          success: false,
          data: [],
          errors: [{ row: 0, error: '文件解析失败，请检查文件格式' }]
        })
      }
    }

    reader.onerror = () => {
      resolve({
        success: false,
        data: [],
        errors: [{ row: 0, error: '文件读取失败' }]
      })
    }

    reader.readAsArrayBuffer(file)
  })
}

# backend/app/api/batch.py
from fastapi import APIRouter, Depends, HTTPException, BackgroundTasks
from typing import List, Optional
from pydantic import BaseModel
from enum import Enum

from app.services.batch_service import BatchService
from app.schemas.common import BatchTaskResponse

router = APIRouter(prefix="/batch", tags=["批量操作"])
batch_service = BatchService()


class UpdateMode(str, Enum):
    IMMEDIATE = "immediate"    # 立即发送
    QUEUED = "queued"          # 队列发送
    SCHEDULED = "scheduled"    # 定时发送


class BatchDisplayUpdateRequest(BaseModel):
    """批量屏幕更新请求"""
    template_id: str                    # 模板ID
    devices: List[dict]                 # 设备列表 [{mac, data: {...}}, ...]
    mode: UpdateMode = UpdateMode.IMMEDIATE  # 更新模式
    concurrency: int = 10               # 并发数
    scheduled_time: Optional[str] = None  # 定时时间(ISO格式)
    auto_retry: bool = True             # 自动重试
    retry_count: int = 3                # 重试次数


class DeviceImportRequest(BaseModel):
    """设备导入请求"""
    file_type: str = "excel"            # excel/csv
    template_fields: List[dict]          # 模板字段定义


@router.post("/display/update", response_model=BatchTaskResponse)
async def batch_display_update(
    request: BatchDisplayUpdateRequest,
    background_tasks: BackgroundTasks
):
    """
    批量更新设备屏幕显示内容
    
    流程说明：
    1. 校验设备和模板数据
    2. 创建批量任务记录
    3. 根据mode决定执行方式：
       - immediate: 立即在后台执行
       - queued: 加入任务队列
       - scheduled: 设置定时任务
    4. 返回任务ID用于跟踪进度
    """
    # 参数校验
    if len(request.devices) == 0:
        raise HTTPException(status_code=400, detail="设备列表不能为空")
    
    if len(request.devices) > 1000:
        raise HTTPException(status_code=400, detail="单次最多支持1000个设备")
    
    # 创建批量任务
    task = await batch_service.create_batch_task(
        task_type="display_update",
        total=len(request.devices),
        params=request.dict()
    )
    
    # 根据模式处理
    if request.mode == UpdateMode.IMMEDIATE:
        # 立即后台执行
        background_tasks.add_task(
            batch_service.execute_batch_display_update,
            task.id,
            request
        )
    elif request.mode == UpdateMode.SCHEDULED:
        # 设置定时任务
        if not request.scheduled_time:
            raise HTTPException(status_code=400, detail="定时模式需要指定时间")
        await batch_service.schedule_task(task.id, request.scheduled_time)
    else:
        # 加入队列
        await batch_service.enqueue_task(task.id)

import asyncio
import json
import base64
from typing import Callable, Dict, Any, Optional
from dataclasses import dataclass
from datetime import datetime

try:
    import paho.mqtt.client as mqtt
except ImportError:
    # 使用异步版本

    USE_ASYNC_MQTT = True


@dataclass
class MqttConfig:
    """MQTT配置"""
    broker: str = "192.144.234.153"
    port: int = 8883
    username: str = ""
    password: str = ""
    api_key: str = ""  # 用户API密钥
    client_id: str = "wifi-esl-manager"
    use_tls: bool = True


class MqttClientWrapper:
    """
    MQTT客户端封装
    
    封装与WIFI标签系统的MQTT通信逻辑，
    包括连接管理、消息发布订阅、Base64编码解码等
    """
    
    def __init__(self, config: MqttConfig):
        self.config = config
        self.client: Optional[Any] = None
        self._connected = False
        self._message_handlers: Dict[str, Callable] = {}
        self._pending_replies: Dict[str, asyncio.Future] = {}
        
    async def connect(self) -> bool:
        """建立MQTT连接"""
        try:
            if USE_ASYNC_MQTT:
                async with MqttClient(
                    hostname=self.config.broker,
                    port=self.config.port,
                    username=self.config.username,
                    password=self.config.password,
                    tls_context=self._get_tls_context()
                ) as client:
                    self.client = client
                    self._connected = True
                    
                    # 订阅所有需要的主题
                    await self._subscribe_topics()
                    
                    # 启动消息监听循环
                    async for message in client.messages:
                        await self._handle_message(message)
            else:
                # 使用同步版本（在单独线程运行）
                self.client = mqtt.Client(client_id=self.config.client_id)
                self.client.username_pw_set(
                    self.config.username, 
                    self.config.password
                )
                
                # TLS配置
                if self.config.use_tls:
                    self.client.tls_set()
                
                # 回调注册
                self.client.on_connect = self._on_connect
                self.client.on_message = self._on_message
                
                # 连接
                self.client.connect(self.config.broker, self.config.port, 60)
                self.client.loop_start()
                
                # 等待连接完成
                await asyncio.sleep(1)
                self._connected = True
            
            print(f"MQTT连接成功: {self.config.broker}:{self.config.port}")
            return True
            
        except Exception as e:
            print(f"MQTT连接失败: {e}")
            self._connected = False
            return False
    
    async def disconnect(self):
        """断开MQTT连接"""
        if self.client:
            if USE_ASYNC_MQTT:
                # 异步版本会在上下文管理器退出时断开
                pass
            else:
                self.client.loop_stop()
                self.client.disconnect()
            self._connected = False
            print("MQTT已断开连接")
    
    async def _subscribe_topics(self):
        """订阅所有需要的MQTT主题"""
        topics = [
            f"/client/{self.config.api_key}/action/online",
            f"/client/{self.config.api_key}/action/offline",
            f"/client/{self.config.api_key}/action/usb_state",
            f"/client/{self.config.api_key}/action/button",
            f"/client/{self.config.api_key}/action/display_reply",
            f"/client/{self.config.api_key}/action/battery_reply",
            f"/client/{self.config.api_key}/action/led_reply",
            f"/client/{self.config.api_key}/action/reboot_reply",
        ]
        
        for topic in topics:
            await self.client.subscribe(topic)
            print(f"已订阅主题: {topic}")
    
    def _get_topic_base(self) -> str:
        """获取主题基础路径"""
        return f"/client/{self.config.api_key}/action"
    
    async def publish_display(self, mac: str, template_data: dict) -> dict:
        """
        发布屏幕更新指令
        
        Args:
            mac: 设备MAC地址
            template_data: 模板数据 {tid, tname, data: {...}}
        
        Returns:
            发布结果
        """
        topic = f"{self._get_topic_base()}/display"
        
        payload = {
            "method": "display",
            "msgId": str(int(datetime.now().timestamp() * 1000)),
            "apiKey": self.config.api_key,
            "version": 1,
            "message": {
                "mac": mac,
                "template": template_data
            }
        }
        
        # Base64编码
        encoded_payload = base64.b64encode(
            json.dumps(payload).encode('utf-8')
        ).decode('utf-8')
        
        # 发布消息
        result = await self.client.publish(topic, encoded_payload)
        
        # 创建Future等待回复
        future = asyncio.Future()
        self._pending_replies[payload["msgId"]] = future
        
        # 设置超时
        try:
            reply = await asyncio.wait_for(future, timeout=30)
            return {"success": True, "reply": reply}
        except asyncio.TimeoutError:
            return {"success": False, "error": "等待回复超时"}
    
    async def publish_led_control(self, mac: str, red: int, green: int, blue: int) -> dict:
        """
        发布LED控制指令
        
        Args:
            mac: 设备MAC地址
            red: 红色 (0-255)
            green: 绿色 (0-255)
            blue: 蓝色 (0-255)
        """
        topic = f"{self._get_topic_base()}/led"
        
        payload = {
            "method": "led",
            "msgId": str(int(datetime.now().timestamp() * 1000)),
            "apiKey": self.config.api_key,
            "version": 1,
            "message": {
                "mac": mac,
                "red": red,
                "green": green,
                "blue": blue
            }
        }
        
        encoded_payload = base64.b64encode(
            json.dumps(payload).encode('utf-8')
        ).decode('utf-8')
        
        await self.client.publish(topic, encoded_payload)
        
        return {"success": True, "message": "LED控制指令已发送"}
    
    async def publish_reboot(self, mac: str) -> dict:
        """发布重启指令"""
        topic = f"{self._get_topic_base()}/reboot"
        
        payload = {
            "method": "reboot",
            "msgId": str(int(datetime.now().timestamp() * 1000)),
            "apiKey": self.config.api_key,
            "version": 1,
            "message": {
                "mac": [mac]
            }
        }
        
        encoded_payload = base64.b64encode(
            json.dumps(payload).encode('utf-8')
        ).decode('utf-8')
        
        await self.client.publish(topic, encoded_payload)
        
        return {"success": True, "message": "重启指令已发送"}
    
    async def query_battery(self, mac: str) -> dict:
        """查询设备电池电量"""
        topic = f"{self._get_topic_base()}/battery"
        
        payload = {
            "method": "battery",
            "msgId": str(int(datetime.now().timestamp() * 1000)),
            "apiKey": self.config.api_key,
            "version": 1,
            "message": {
                "mac": mac
            }
        }
        
        encoded_payload = base64.b64encode(
            json.dumps(payload).encode('utf-8')
        ).decode('utf-8')
        
        future = asyncio.Future()
        self._pending_replies[payload["msgId"]] = future
        
        await self.client.publish(topic, encoded_payload)
        
        try:
            reply = await asyncio.wait_for(future, timeout=15)
            return {"success": True, "voltage": reply.get("voltage")}
        except asyncio.TimeoutError:
            return {"success": False, "error": "查询超时"}
    
    def register_handler(self, action: str, handler: Callable):
        """注册消息处理器"""
        self._message_handlers[action] = handler
    
    async def _handle_message(self, message):
        """处理接收到的MQTT消息"""
        try:
            # Base64解码
            decoded = base64.b64decode(message.payload.decode())
            data = json.loads(decoded)
            
            # 根据主题类型分发处理
            topic_parts = message.topic.split("/")
            action = topic_parts[-1]
            
            # 检查是否是回复消息
            if action.endswith("_reply") and "msgId" in data:
                msg_id = data["msgId"]
                if msg_id in self._pending_replies:
                    future = self._pending_replies.pop(msg_id)
                    if not future.done():
                        future.set_result(data)
            
            # 调用注册的处理器
            handler = self._message_handlers.get(action)
            if handler:
                await handler(data)
                
        except Exception as e:
            print(f"处理MQTT消息出错: {e}")

# backend/app/services/batch_service.py
import asyncio
from typing import List, Dict, Any, Optional
from datetime import datetime
from dataclasses import dataclass, field
from enum import Enum
import json

from app.core.mqtt_client import mqtt_client
from app.models.task import BatchTask, TaskItem, TaskStatus


class TaskStatusEnum(Enum):
    PENDING = "pending"      # 待执行
    RUNNING = "running"      # 执行中
    COMPLETED = "completed"  # 已完成
    PARTIAL = "partial"      # 部分完成
    FAILED = "failed"        # 失败
    CANCELLED = "cancelled"  # 已取消


@dataclass
class BatchTaskInfo:
    """批量任务信息"""
    id: str
    task_type: str
    status: TaskStatusEnum
    total: int
    success_count: int = 0
    failed_count: int = 0
    created_at: datetime = field(default_factory=datetime.now)
    started_at: Optional[datetime] = None
    completed_at: Optional[datetime] = None
    items: List[Dict[str, Any]] = field(default_factory=list)


class BatchService:
    """
    批量操作服务
    
    处理批量设备控制的调度、执行和结果收集
    """
    
    def __init__(self):
        self._tasks: Dict[str, BatchTaskInfo] = {}
        self._semaphore: Optional[asyncio.Semaphore] = None
    
    async def create_batch_task(
        self, 
        task_type: str, 
        total: int,
        params: Dict[str, Any]
    ) -> BatchTaskInfo:
        """创建批量任务"""
        task_id = f"task_{int(datetime.now().timestamp())}"
        
        task = BatchTaskInfo(
            id=task_id,
            task_type=task_type,
            status=TaskStatusEnum.PENDING,
            total=total,
            params=params
        )
        
        self._tasks[task_id] = task
        return task
    
    async def execute_batch_display_update(
        self, 
        task_id: str, 
        request: Dict[str, Any]
    ):
        """
        执行批量屏幕更新任务
        
        使用信号量控制并发数，逐个或并发地发送MQTT消息
        """
        task = self._tasks.get(task_id)
        if not task:
            return
        
        task.status = TaskStatusEnum.RUNNING
        task.started_at = datetime.now()
        
        # 设置并发控制
        concurrency = request.get("concurrency", 10)
        self._semaphore = asyncio.Semaphore(concurrency)
        
        devices = request.get("devices", [])
        template_id = request.get("template_id")
        auto_retry = request.get("auto_retry", True)
        retry_count = request.get("retry_count", 3)
        
        # 创建协程任务列表
        tasks = []
        for idx, device in enumerate(devices):
            mac = device.get("mac")
            data = device.get("data", {})
            
            task_item = {
                "index": idx,
                "mac": mac,
                "data": data,
                "status": "pending",
                "retry_count": 0,
                "error": None
            }
            task.items.append(task_item)
            
            tasks.append(self._update_single_device(
                task_id=task_id,
                item_index=idx,
                mac=mac,
                template_id=template_id,
                template_data=data,
                auto_retry=auto_retry,
                max_retries=retry_count
            ))
        
        # 并发执行所有任务
        results = await asyncio.gather(*tasks, return_exceptions=True)
        
        # 统计结果
        success = sum(1 for r in results if r is True)
        failed = sum(1 for r in results if r is not True)
        
        task.success_count = success
        task.failed_count = failed
        task.completed_at = datetime.now()
        
        # 设置最终状态
        if failed == 0:
            task.status = TaskStatusEnum.COMPLETED
        elif success == 0:
            task.status = TaskStatusEnum.FAILED
        else:
            task.status = TaskStatusEnum.PARTIAL
        
        print(f"批量任务 {task_id} 完成: 成功={success}, 失败={failed}")
    
    async def _update_single_device(
        self,
        task_id: str,
        item_index: int,
        mac: str,
        template_id: str,
        template_data: Dict[str, Any],
        auto_retry: bool,
        max_retries: int
    ) -> bool:
        """
        更新单个设备的屏幕显示（带重试机制）
        """
        task = self._tasks.get(task_id)
        
        async with self._semaphore:
            retries = 0
            last_error = None
            
            while retries <= max_retries:
                try:
                    # 构造模板数据
                    full_template_data = {
                        "tid": template_id,
                        "tname": "",  # 可以从模板缓存获取
                        "data": template_data
                    }
                    
                    # 发送MQTT消息
                    result = await mqtt_client.publish_display(
                        mac=mac,
                        template_data=full_template_data
                    )
                    
                    if result.get("success"):
                        # 更新任务项状态
                        if task and item_index < len(task.items):
                            task.items[item_index]["status"] = "success"
                        return True
                    else:
                        last_error = result.get("error", "未知错误")
                        
                except Exception as e:
                    last_error = str(e)
                
                # 重试逻辑
                if auto_retry and retries < max_retries:
                    retries += 1
                    task.items[item_index]["retry_count"] = retries
                    await asyncio.sleep(1 * retries)  # 递增延迟
                else:
                    break
            
            # 标记失败
            if task and item_index < len(task.items):
                task.items[item_index]["status"] = "failed"
                task.items[item_index]["error"] = last_error
            
            return False
    
    async def get_task(self, task_id: str) -> Optional[BatchTaskInfo]:
        """获取任务信息"""
        return self._tasks.get(task_id)
    
    async def get_task_results(
        self, 
        task_id: str, 
        page: int = 1, 
        page_size: int = 50
    ) -> Dict[str, Any]:
        """获取任务执行结果的分页数据"""
        task = self._tasks.get(task_id)
        if not task:
            return {"items": [], "total": 0}
        
        start = (page - 1) * page_size
        end = start + page_size
        items = task.items[start:end]
        
        return {
            "items": items,
            "total": len(task.items),
            "page": page,
            "page_size": page_size,
            "summary": {
                "success": task.success_count,
                "failed": task.failed_count,
                "status": task.status.value
            }
        }
    
    async def cancel_task(self, task_id: str) -> bool:
        """取消正在执行的任务"""
        task = self._tasks.get(task_id)
        if task and task.status == TaskStatusEnum.RUNNING:
            task.status = TaskStatusEnum.CANCELLED
            # TODO: 实现取消正在运行的协程
            return True
        return False

### 5.2 前端实时状态Hook

```typescript
// composables/useRealtimeStatus.ts
import { ref, onMounted, onUnmounted } from 'vue'
import { useMqttStore } from '@/stores/mqtt'
import { useDeviceStore } from '@/stores/device'

interface RealtimeDeviceStatus {
  mac: string
  online: boolean
  voltage?: number
  rssi?: number
  usbState?: number
  lastUpdate: Date
  displayResult?: {
    success: boolean
    timestamp: Date
  }
}

export function useRealtimeStatus() {
  const mqttStore = useMqttStore()
  const deviceStore = useDeviceStore()
  
  const realtimeStatusMap = ref<Map<string, RealtimeDeviceStatus>>(new Map())
  const isConnecting = ref(false)
  
  /**
   * 初始化实时状态监听
   */
  function initRealtimeMonitor() {
    isConnecting.value = true
    
    // 监听MQTT连接状态
    mqttStore.$onConnectionChange((connected) => {
      if (connected) {
        console.log('MQTT已连接，开始接收实时状态')
      } else {
        console.warn('MQTT连接断开，尝试重连...')
      }
    })
    
    // 监听设备上线
    mqttStore.subscribe('/client/${apiKey}/action/online', (payload) => {
      const { mac, msgId } = payload
      updateStatus(mac, { online: true, lastUpdate: new Date() })
      deviceStore.updateDeviceOnline(mac, true)
    })
    
    // 监听设备下线
    mqttStore.subscribe('/client/${apiKey}/action/offline', (payload) => {
      const { mac, msgId } = payload
      updateStatus(mac, { online: false, lastUpdate: new Date() })
      deviceStore.updateDeviceOnline(mac, false)
    })
    
    // 监听USB状态变化
    mqttStore.subscribe('/client/${apiKey}/action/usb_state', (payload) => {
      const { mac, state } = payload
      updateStatus(mac, { usbState: state, lastUpdate: new Date() })
    })
    
    // 监听电池电量回复
    mqttStore.subscribe('/client/${apiKey}/action/battery_reply', (payload) => {
      const { mac, voltage } = payload
      updateStatus(mac, { voltage, lastUpdate: new Date() })
      deviceStore.updateDeviceField(mac, 'voltage', voltage)
    })
    
    // 监听屏幕更新回复
    mqttStore.subscribe('/client/${apiKey}/action/display_reply', (payload) => {
      const { mac, result, msgId } = payload
      updateStatus(mac, { 
        displayResult: { 
          success: result === 200, 
          timestamp: new Date() 
        },
        lastUpdate: new Date()
      })
      deviceStore.updateDisplayResult(mac, result === 200)
      
      // 触发通知
      showNotification(mac, result === 200)
    })
    
    isConnecting.value = false
  }
  
  /**
   * 更新设备实时状态
   */
  function updateStatus(mac: string, updates: Partial<RealtimeDeviceStatus>) {
    const current = realtimeStatusMap.value.get(mac) || {
      mac,
      online: false,
      lastUpdate: new Date()
    }
    realtimeStatusMap.value.set(mac, { ...current, ...updates })
  }
  
  /**
   * 获取设备最新状态
   */
  function getStatus(mac: string): RealtimeDeviceStatus | undefined {
    return realtimeStatusMap.value.get(mac)
  }
  
  /**
   * 检查状态是否过期（超过60秒无更新视为可能离线）
   */
  function isStale(mac: string, thresholdSeconds = 60): boolean {
    const status = realtimeStatusMap.value.get(mac)
    if (!status) return true
    
    const elapsed = (Date.now() - status.lastUpdate.getTime()) / 1000
    return elapsed > thresholdSeconds
  }
  
  /**
   * 显示操作结果通知
   */
  function showNotification(mac: string, success: boolean) {
    // 这里可以使用ElNotification或自定义通知组件
    const title = success ? '屏幕更新成功' : '屏幕更新失败'
    const message = `设备 ${mac} ${success ? '已成功' : '未能'}完成屏幕更新`
    const type = success ? 'success' : 'error'
    
    // 发送自定义事件，由各组件自行决定如何展示
    window.dispatchEvent(new CustomEvent('notification', {
      detail: { title, message, type, mac }
    }))
  }
  
  onMounted(() => {
    initRealtimeMonitor()
  })
  
  onUnmounted(() => {
    // 清理资源
    realtimeStatusMap.value.clear()
  })
  
  return {
    realtimeStatusMap,
    isConnecting,
    getStatus,
    isStale,
    initRealtimeMonitor
  }
}
```

## 十一、数据库可切换架构设计

### 11.1 设计目标

本系统采用 **SQLAlchemy ORM** 作为数据库访问层，实现 **数据库无关性**，支持在 SQLite、MySQL、PostgreSQL 等数据库之间无缝切换，业务代码无需修改。

### 11.3 数据库连接配置

#### 核心配置文件（database.py）

```python
# backend/app/core/database.py
from sqlalchemy.ext.asyncio import create_async_engine, AsyncSession, async_sessionmaker
from sqlalchemy.orm import DeclarativeBase
from app.config import settings


class Base(DeclarativeBase):
    """所有ORM模型的基类"""
    pass


# 根据配置创建引擎（自动适配不同数据库）
def create_db_engine():
    """
    创建数据库引擎
    
    支持的数据库URL格式：
    - SQLite:   sqlite+aiosqlite:///./data/esl_manager.db
    - MySQL:    mysql+aiomysql://user:password@localhost:3306/esl_manager
    - PostgreSQL: postgresql+asyncpg://user:password@localhost:5432/esl_manager
    """
    
    db_url = settings.DATABASE_URL
    
    # 不同数据库的优化参数
    if db_url.startswith("sqlite"):
        # SQLite 特殊配置（单线程、外键检查）
        engine = create_async_engine(
            db_url,
            echo=settings.DEBUG,
            connect_args={"check_same_thread": False}
        )
    elif db_url.startswith("mysql") or db_url.startswith("postgresql"):
        # MySQL/PostgreSQL 连接池配置
        engine = create_async_engine(
            db_url,
            echo=settings.DEBUG,
            pool_size=20,              # 连接池大小
            max_overflow=10,           # 最大溢出连接数
            pool_recycle=3600,         # 连接回收时间(秒) - MySQL建议设置
            pool_pre_ping=True        # 连接前健康检查
        )
    else:
        raise ValueError(f"不支持的数据库类型: {db_url.split(':')[0]}")
    
    return engine


# 创建全局引擎实例
engine = create_db_engine()

# 创建会话工厂
async_session_factory = async_sessionmaker(
    engine,
    class_=AsyncSession,
    expire_on_commit=False
)


async def get_db() -> AsyncSession:
    """FastAPI依赖注入：获取数据库会话"""
    async with async_session_factory() as session:
        try:
            yield session
            await session.commit()
        except Exception:
            await session.rollback()
            raise
        finally:
            await session.close()


async def init_db():
    """初始化数据库（创建表结构）"""
    async with engine.begin() as conn:
        await conn.run_sync(Base.metadata.create_all)
```

### 11.4 跨数据库兼容的模型定义

```python
# backend/app/models/device.py
from sqlalchemy import Column, String, Integer, Boolean, DateTime, Text, JSON, ForeignKey, Index
from sqlalchemy.orm import relationship
from datetime import datetime
import uuid

from app.core.database import Base


class Device(Base):
    """
    设备模型 - 跨数据库兼容设计
    
    注意事项：
    1. 字符串字段必须指定长度（MySQL要求）
    2. 使用JSON类型而非特定数据库的JSONB（保持通用性）
    3. 时间统一使用datetime.utcnow（避免时区问题）
    4. 主键使用String UUID（便于分布式扩展）
    """
    __tablename__ = "devices"
    
    # ========== 主键 ==========
    id = Column(String(36), primary_key=True, default=lambda: str(uuid.uuid4()))
    
    # ========== 基本信息 ==========
    mac = Column(String(17), unique=True, nullable=False, index=True, comment="设备MAC地址")
    ip = Column(String(45), nullable=True, comment="IP地址（支持IPv6长度）")
    name = Column(String(100), nullable=True, comment="自定义设备名称")
    
    # ========== 实时状态（来自MQTT推送）==========
    is_online = Column(Boolean, default=False, index=True, comment="在线状态")
    voltage = Column(Integer, nullable=True, comment="电池电量(mV)")
    rssi = Column(Integer, nullable=True, comment="信号强度(dBm)")
    usb_state = Column(Integer, nullable=True, comment="USB状态 0/1")
    last_seen = Column(DateTime, nullable=True, comment="最后上线时间")
    
    # ========== WIFI/MQTT配置（JSON存储）==========
    station_config = Column(JSON, nullable=True, comment="WIFI站点配置 {ssid, password}")
    mqtt_config = Column(JSON, nullable=True, comment="MQTT配置 {broker, port, username}")
    
    # ========== 设备属性 ==========
    device_type = Column(String(20), nullable=True, comment="设备类型 esl_29/esl_43/tag")
    screen_type = Column(String(20), nullable=True, comment="屏幕类型 bw/bwr/color")
    sn = Column(String(50), nullable=True, comment="序列号")
    sw_version = Column(Integer, nullable=True, comment="软件版本号")
    hw_version = Column(Integer, nullable=True, comment="硬件版本号")
    
    # ========== 关联关系 ==========
    user_id = Column(String(36), ForeignKey("users.id", ondelete="SET NULL"), nullable=True)
    product_id = Column(String(36), ForeignKey("products.id", ondelete="SET NULL"), nullable=True)
    
    # 关系定义（lazy='selectin' 避免N+1查询问题）
    user = relationship("User", backref="devices", lazy="selectin")
    product = relationship("Product", backref="devices")
    groups = relationship("DeviceGroup", secondary="device_group_mapping", backref="devices")
    
    # ========== 时间戳 ==========
    created_at = Column(DateTime, default=datetime.utcnow, comment="创建时间")
    updated_at = Column(DateTime, default=datetime.utcnow, onupdate=datetime.utcnow, comment="更新时间")
    
    # ========== 索引优化（所有数据库通用语法）==========
    __table_args__ = (
        Index('idx_device_online', 'is_online'),                    # 单列索引：按在线状态筛选
        Index('idx_device_user_online', 'user_id', 'is_online'),   # 复合索引：用户+状态联合查询
        Index('idx_device_updated', 'updated_at'),                  # 用于增量同步
        {'comment': '设备信息表'}                                    # 表注释
    )
    
    def to_dict(self):
        """序列化为字典"""
        return {
            'id': self.id,
            'mac': self.mac,
            'ip': self.ip,
            'name': self.name,
            'is_online': self.is_online,
            'voltage': self.voltage,
            'rssi': self.rssi,
            'device_type': self.device_type,
            'screen_type': self.screen_type,
            'created_at': self.created_at.isoformat() if self.created_at else None,
        }
```
